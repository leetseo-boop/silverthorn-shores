# Bluehost 301 redirects — Silverthorn old site → new domain
# Paste these rules into the .htaccess file at the DOCUMENT ROOT of the old Bluehost account.
# Each old path maps to its matching page on silverthornresort.com (the new Lovable site).
#
# IMPORTANT: These rules only affect HTTP/web traffic. They do NOT touch MX, SPF, DKIM,
# or DMARC records, so reservations1@silverthornresort.com and all other mailboxes
# continue to work exactly as before.

RewriteEngine On

# Home
Redirect 301 /index.html https://silverthornresort.com/
Redirect 301 /home https://silverthornresort.com/

# Final catch-all: anything else on the old site → new homepage
# (Keep this LAST — Apache evaluates rules top-to-bottom and this one matches everything.)
RewriteRule ^(.*)$ https://silverthornresort.com/ [R=301,L]
